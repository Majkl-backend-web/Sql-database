CREATE TABLE Client(
ClientId int PRIMARY KEY,

CREATE TABLE Animal(
AnimalId int PRIMARY KEY,
AnimalName varchar(50),
Species varchar(50),
Age int,
Dangerousity varchar(50),
EmployeeId int
CONSTRAINT FK_employee_animal FOREIGN KEY (EmployeeId) REFERENCES Employee(EmployeeId),
);

CREATE TABLE AnimalEmployee(
AnimalId int,
EmployeeId int,
CONSTRAINT FK_animalEmployee_animal FOREIGN KEY (AnimalId) REFERENCES Animal(AnimalId),
CONSTRAINT FK_animalEmployee_employee FOREIGN KEY (EmployeeId) REFERENCES Person(PersonId)
);

CREATE TABLE Booking(
BookingId int PRIMARY KEY,
FromDate date,
ToDate date,
PaymentId int,
AnimalId int,
ClientId int,
CONSTRAINT FK_payment_booking FOREIGN KEY (PaymentId) REFERENCES Payment(PaymentId),
CONSTRAINT FK_animal_booking FOREIGN KEY (AnimalId) REFERENCES Animal(AnimalId),
CONSTRAINT FK_person_booking FOREIGN KEY (ClientId) REFERENCES Client(ClientId)
);
